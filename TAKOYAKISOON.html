function loadCSV(event, session) {

  const file = event.target.files[0];

  if (!file) return;


  setStatus('⟳ 파일 읽는 중...', 'var(--muted)');


  // 키움은 EUC-KR, 나무(미국)는 UTF-8 시도 후 EUC-KR fallback

  const enc = session === 'kr' ? 'EUC-KR' : 'UTF-8';

  readFileWithEncoding(file, enc)

    .then(text => processCSVText(text, session, file.name))

    .catch(() => {

      // UTF-8 실패 시 EUC-KR 재시도

      return readFileWithEncoding(file, 'EUC-KR')

        .then(text => processCSVText(text, session, file.name));

    })

    .catch(err => setStatus('⚠ ' + err.message, 'var(--red)'));

}


function readFileWithEncoding(file, encoding) {

  return new Promise((resolve, reject) => {

    const reader = new FileReader();

    reader.onload  = e => resolve(e.target.result);

    reader.onerror = () => reject(new Error('파일 읽기 실패'));

    reader.readAsText(file, encoding);

  });

}


// ── CSV 텍스트 처리 ──

function processCSVText(text, session, filename) {

  // 파일명으로 증권사 자동 감지

  const broker = detectBroker(text, filename, session);

  const trades = broker === 'kiwoom'

    ? parseKiwoomCSV(text)

    : parseNamuCSV(text);


  if (trades.length === 0) {

    setStatus('⚠ 체결 데이터를 찾을 수 없습니다', 'var(--amber)');

    return;

  }


  // 전역 데이터 교체

  if (session === 'kr') { krT.length = 0; krT.push(...trades); }

  else                  { usT.length = 0; usT.push(...trades); }


  // 저장 & 화면 갱신

  const now = new Date().toLocaleString('ko-KR');

  localStorage.setItem(`trades_${session}`,      JSON.stringify(trades));

  localStorage.setItem(`trades_${session}_date`, now);


  refreshDashboard(session, trades);

  setConnStatus(session, true, trades.length);


  const syncEl = document.getElementById('sync-time');

  if (syncEl) syncEl.textContent = `마지막 로드: ${now}`;


  setStatus(`✓ ${trades.length}건 로드 완료 (${broker})`, 'var(--green)');

  setSession(session);


  // 입력 초기화 (같은 파일 재업로드 가능하게)

  event && event.target && (event.target.value = '');

}


// ── 증권사 자동 감지 ──

function detectBroker(text, filename, session) {

  if (session === 'kr') {

    if (filename.includes('키움') || filename.toLowerCase().includes('kiwoom')) return 'kiwoom';

    if (text.includes('영웅문') || text.includes('체결잔고') || text.includes('매수단가')) return 'kiwoom';

    return 'kiwoom'; // KR 기본값

  } else {

    return 'namu'; // US 기본값

  }

}

// 키움증권 CSV 파서 (영웅문4 체결잔고 0312 기준)

// ══════════════════════════════════════════════════

function parseKiwoomCSV(text) {

  const trades = [];

  const lines  = text.split(/\r?\n/).map(l => l.trim()).filter(l => l && !l.startsWith('#'));


  // 헤더 행 탐색 (종목명·체결·손익 중 하나라도 포함)

  const headerKeywords = ['종목명','종목코드','매매구분','체결','손익','매수단가'];

  let headerIdx = lines.findIndex(l => headerKeywords.some(k => l.includes(k)));

  if (headerIdx === -1) headerIdx = 0;


  const headers = splitCSVLine(lines[headerIdx]);


  // 컬럼 인덱스 사전 매핑 (유연하게)

  const COL = {};

  const colMap = {

    name  : ['종목명','종목'],

    buyPx : ['매수단가','평균매수가','매수가'],

    sellPx: ['매도단가','평균매도가','매도가','청산가'],

    pnl   : ['실현손익','손익금액','실현수익'],

    qty   : ['수량','체결수량','매수수량'],

    type  : ['매매구분','구분'],

    date  : ['일자','체결일자','날짜'],

    time  : ['체결시간','시간'],

  };

  Object.entries(colMap).forEach(([key, candidates]) => {

    const idx = headers.findIndex(h => candidates.includes(h));

    COL[key] = idx;

  });


  for (let i = headerIdx + 1; i < lines.length; i++) {

    const cols = splitCSVLine(lines[i]);

    if (cols.length < 4) continue;


    const get = key => COL[key] >= 0 ? (cols[COL[key]] || '') : '';


    const name   = get('name');

    const buyPx  = parseNum(get('buyPx'));

    const sellPx = parseNum(get('sellPx'));

    const pnl    = parseNum(get('pnl'));

    const qty    = parseInt(get('qty').replace(/,/g,'')) || 1;

    const rawType = get('type');

    const date   = get('date');

    const time   = get('time');


    // 유효성 검사

    if (!name || name.length < 2) continue;

    if (buyPx === 0 && sellPx === 0) continue;


    const win  = pnl > 0;

    const type = rawType.includes('선물') ? 'fut'

               : rawType.includes('옵션') ? 'opt'

               : rawType.includes('매도') ? 'short'

               : 'long';


    const absK = Math.abs(pnl);

    trades.push({

      l    : name,

      entry: buyPx  || sellPx,

      exit : sellPx || buyPx,

      pnl  : pnl,

      ps   : `${win?'+':''}${pnl.toLocaleString()}원`,

      pl   : `${win?'+':''}${(absK>=10000 ? Math.round(absK/10000)+'만' : absK.toLocaleString())}`,

      w    : win,

      qty  : qty,

      date : date,

      time : time,

      type : type,

    });

  }


  return trades;

}

// 나무증권(미국) CSV 파서

// ══════════════════════════════════════════════════

function parseNamuCSV(text) {

  const trades = [];

  const lines  = text.split(/\r?\n/).map(l => l.trim()).filter(l => l && !l.startsWith('#'));


  const headerKeywords = ['종목','매수','체결','Symbol','Ticker','Buy','Sell','P&L'];

  let headerIdx = lines.findIndex(l => headerKeywords.some(k => l.includes(k)));

  if (headerIdx === -1) headerIdx = 0;


  const headers = splitCSVLine(lines[headerIdx]);


  const COL = {};

  const colMap = {

    name  : ['종목명','Symbol','Ticker','종목'],

    buyPx : ['매수가','Buy Price','진입가','Entry'],

    sellPx: ['매도가','Sell Price','청산가','Exit'],

    pnl   : ['손익','P&L','실현손익','Realized'],

    qty   : ['수량','Qty','Quantity'],

    date  : ['일자','Date','날짜'],

    cur   : ['통화','Currency'],

  };

  Object.entries(colMap).forEach(([key, candidates]) => {

    const idx = headers.findIndex(h => candidates.some(c => h.includes(c)));

    COL[key] = idx;

  });


  for (let i = headerIdx + 1; i < lines.length; i++) {

    const cols = splitCSVLine(lines[i]);

    if (cols.length < 3) continue;


    const get = key => COL[key] >= 0 ? (cols[COL[key]] || '') : '';


    const name   = get('name').replace(/"/g,'');

    const buyPx  = parseNum(get('buyPx').replace(/[$,]/g,''));

    const sellPx = parseNum(get('sellPx').replace(/[$,]/g,''));

    const pnl    = parseNum(get('pnl').replace(/[$,+]/g,''));

    const qty    = parseInt(get('qty').replace(/,/g,'')) || 1;

    const date   = get('date');

    const isUSD  = /^[A-Z]{1,5}[0-9]?$/.test(name) || get('cur') === 'USD';


    if (!name || name.length < 1) continue;

    if (buyPx === 0 && sellPx === 0 && pnl === 0) continue;


    const win = pnl >= 0;

    const absP = Math.abs(pnl);

    trades.push({

      l    : name,

      entry: buyPx  || sellPx,

      exit : sellPx || buyPx,

      pnl  : pnl,

      ps   : `${win?'+':''}${isUSD ? '$' : ''}${absP.toLocaleString()}`,

      pl   : `${win?'+':''}${isUSD ? '$' : ''}${absP.toLocaleString()}`,

      w    : win,

      qty  : qty,

      date : date,

      type : name.includes('선물') || name.includes('ES') || name.includes('NQ') ? 'fut'

           : name.includes('C ') || name.includes('P ') ? 'opt'

           : 'long',

    });

  }


  return trades;

}


// ── CSV 라인 파서 (따옴표 안 쉼표 처리) ──

function splitCSVLine(line) {

  const result = [];

  let cur = '', inQ = false;

  for (let i = 0; i < line.length; i++) {

    const c = line[i];

    if (c === '"') { inQ = !inQ; continue; }

    if (c === ',' && !inQ) { result.push(cur.trim()); cur = ''; continue; }

    cur += c;

  }

  result.push(cur.trim());

  return result;

}


// ── 숫자 파싱 (한/영 혼용) ──

function parseNum(s) {

  if (!s) return 0;

  const n = parseFloat(String(s).replace(/,/g,'').replace(/[원₩]/g,'').trim());

  return isNaN(n) ? 0 : n;

}

function refreshDashboard(session, trades) {

  const wins   = trades.filter(t => t.w);

  const losses = trades.filter(t => !t.w);

  const total  = trades.reduce((s, t) => s + t.pnl, 0);

  const avgW   = wins.length   ? wins.reduce((s,t) => s + t.pnl, 0) / wins.length : 0;

  const avgL   = losses.length ? Math.abs(losses.reduce((s,t) => s + t.pnl, 0) / losses.length) : 1;

  const ratio  = avgL > 0 ? (avgW / avgL).toFixed(1) : '∞';

  const wr     = trades.length ? ((wins.length / trades.length) * 100).toFixed(1) + '%' : '0%';

  const isKR   = session === 'kr';


  // KPI 업데이트

  const dash = document.getElementById(`dash-${session}`);

  if (!dash) return;


  const kvals = dash.querySelectorAll('.kval');

  const ksubs = dash.querySelectorAll('.ksub');


  if (kvals[0]) {

    kvals[0].textContent = `${total >= 0 ? '+' : ''}${total.toLocaleString()}${isKR ? '' : ' USD'}`;

    kvals[0].className   = `kval ${total >= 0 ? 'up' : 'dn'}`;

  }

  if (kvals[1]) { kvals[1].textContent = wr; kvals[1].className = 'kval n'; }

  if (kvals[2]) { kvals[2].textContent = trades.length; }

  if (ksubs[2]) { ksubs[2].textContent = `익절 ${wins.length} / 손절 ${losses.length}`; }


  // 차트 재렌더

  const eeId  = session + '-ee';

  const pnlId = session + '-pnl';

  Chart.getChart(eeId)?.destroy();

  Chart.getChart(pnlId)?.destroy();

  makeEE(eeId, trades);

  makePnl(pnlId, trades, isKR ? '원' : 'USD');


  // 대시보드 미니 테이블 (최신 5건)

  const tbody = dash.querySelector('tbody');

  if (tbody) {

    const typeLabel = { long:'매수', short:'매도', fut:'선물', opt:'옵션' };

    tbody.innerHTML = trades.slice(0, 5).map(t => `

        <td class="${t.w ? 'pu' : 'pd'}">${t.ps}</td>

      </tr>`).join('');

  }

}

// Google Drive 공유 링크 → 다운로드 URL 변환

function gdriveShareToDownload(shareUrl) {

  // 형식: https://drive.google.com/file/d/FILE_ID/view?...

  const m = shareUrl.match(/\/d\/([a-zA-Z0-9_-]+)/);

  if (!m) throw new Error('올바른 Google Drive 공유 링크가 아닙니다');

  const fileId = m[1];

  // CORS 우회: Google Drive 직접 export URL 사용

  return `https://drive.google.com/uc?export=download&id=${fileId}`;

}


async function fetchGDriveFile() {

  const url  = document.getElementById('gdrive-url').value.trim();

  const sess = document.querySelector('input[name="gdrive-sess"]:checked').value;

  const btn  = document.getElementById('gdrive-confirm');


  if (!url) {

    document.getElementById('gdrive-url').style.borderColor = 'var(--red)';

    return;

  }


  btn.textContent = '⟳ 다운로드 중...';

  btn.disabled = true;


  try {

    const dlUrl = gdriveShareToDownload(url);


    // CORS 이슈가 있을 경우 CORS 프록시 사용

    const proxyUrl = `https://corsproxy.io/?${encodeURIComponent(dlUrl)}`;


    const res = await fetch(proxyUrl);

    if (!res.ok) throw new Error(`다운로드 실패 (${res.status}). 파일 공유 설정을 확인하세요.`);


    const text = await res.text();


    document.getElementById('gdrive-modal').remove();

    processCSVText(text, sess, url);


  } catch(err) {

    btn.textContent = '불러오기';

    btn.disabled = false;

    document.getElementById('gdrive-url').style.borderColor = 'var(--red)';

    setStatus('⚠ ' + err.message, 'var(--red)');


    // 에러 인라인 표시

    const errDiv = document.createElement('div');

    errDiv.style.cssText = 'font-size:11px;color:var(--red);margin-top:8px;padding:8px;background:rgba(248,113,113,0.08);border-radius:6px';

    errDiv.textContent = '⚠ ' + err.message;

    document.getElementById('gdrive-confirm').parentElement.before(errDiv);

  }

}
